trigger_to_script_bitsum = {
    value = scope:tmp_testbit
    if = {
        limit = {
            exists = scope:tmp_numbit
        }
        add = scope:tmp_numbit
    }
}

cpm_loyalist_academics = {
    if = {
        limit = {
            save_temporary_trigger_breakpoint_as = {
                TRIGGER = cpm_strig_loyalists
                NAME = tmp
				POPTYPE = academics
            }
        }
        add = scope:tmp
    }
}
